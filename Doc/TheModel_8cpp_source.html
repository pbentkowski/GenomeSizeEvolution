44 double envGenerate(double turbulence, double twice_turbulence, double RNG,
45  double env){
46  double delta_env;
47  double tmp_env;
48  delta_env = twice_turbulence * RNG - turbulence;
49  tmp_env = env + delta_env;
50  if (tmp_env < -1.0) {
51  env = -2.0 - (env + delta_env);
52  }
53  else if (tmp_env > 1.0) {
54  env = 2.0 - (env + delta_env);
55  } else {
56  env = tmp_env;
57  }
58  return env;
59 }
