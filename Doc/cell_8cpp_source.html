113 void Cell::mutateCell(double Env, double duplication_probability,
114  double deletion_probability, double mutate_threshold, int minimum_genotype_size,
115  double not_super_gene_param, double gene_space_width, double environment_resolution,
116  double one_over_total_env_space_surf) {
117  // calling and initialising random number generator
118  rngEngine* p_rngEngine = rngEngine::getInstance();
119  Tagging_system* pTagging_system = Tagging_system::getInstance();
120  // this is Cell's class data field, if not mutated will be false
121  TestIfMutated = false;
122  // calculating what's the probability of any duplication and if positive
123  // entering duplication method
124  if (p_rngEngine->rngMTgetDauble() < (1.0 - gsl_pow_int(1.0 - duplication_probability,
125  (int) CellsGenotype.GenotypeValues.size()))) {
126  CellsGenotype.duplicateGene(duplication_probability);
127  TestIfMutated = true;
128  }
129  // calculating what's the probability of any deletion and if positive entering
130  // deletion method
131  if (p_rngEngine->rngMTgetDauble() < (1.0 - gsl_pow_int(1.0 - deletion_probability,
132  (int) CellsGenotype.GenotypeValues.size()))) {
133  CellsGenotype.deleteGenes(deletion_probability, minimum_genotype_size);
134  TestIfMutated = true;
135  }
136  // calculating what's the probability of ANY single-gene mutation appearing and if
137  // positive, then entering mutation method
138  if (p_rngEngine->rngMTgetDauble() > gsl_pow_int(1.0 - mutate_threshold,
139  (int) CellsGenotype.GenotypeValues.size())) {
140  CellsGenotype.mutateGenes(mutate_threshold, not_super_gene_param,
141  gene_space_width);
142  TestIfMutated = true;
143  }
144  if (TestIfMutated == true) {
145  //evaluateGrowthRateFromGenotypeOnePointEnv(Env)
146  // wasn't necessary to call it here
147  OwnTag = pTagging_system->getTag();
148  }
149  CellsGenotype.evaluateAdaptiveLandcape(environment_resolution,
150  one_over_total_env_space_surf);
151 }

162 void Cell::evaluateGrowthRateFromGenotypeOnePointEnv(double Env) {
163  // measuring genome size
164  int GenomeSize = (int) CellsGenotype.GenotypeValues.size();
165  gsl_vector * TempVector = gsl_vector_alloc(GenomeSize);
166  for (int j = 0; j < GenomeSize; j++) {
167  // calculating growth rate for a given gene for a current Environment value
168  gsl_vector_set(TempVector, j, CellsGenotype.calculateGrowthFromOneGene(j, Env));
169  }
170  // picking the maximal value for given Env conditions
171  CellsGrossGrowthRateAtTheMoment = gsl_vector_max(TempVector);
172  gsl_vector_free(TempVector);
173 }

224 void Cell::evaluateInternalResourcesCost(double gene_replication_cost,
225  int core_genes, double metabolic_costs) {
226  double GenomeSize = (double) CellsGenotype.GenotypeValues.size();
227  double recentCostTemp;
228  // sum of cost of living in one iteration
229  recentCostTemp = -gene_replication_cost * ((double) core_genes \
230  + GenomeSize + GenomeSize * GenomeSize) - metabolic_costs;
231  if (recentCostTemp < -CellsResource) {
232  recentLoss = -CellsResource;
233  CellsResource = 0.0;
234  }
235  if (recentCostTemp >= -CellsResource) {
236  recentLoss = recentCostTemp;
237  CellsResource = CellsResource + recentCostTemp;
238  }
239 }
