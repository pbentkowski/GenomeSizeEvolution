00162 void Cell::evaluateGrowthRateFromGenotypeOnePointEnv(double Env) {
00163     // measuring genome size
00164     int GenomeSize = (int) CellsGenotype.GenotypeValues.size();
00165     gsl_vector * TempVector = gsl_vector_alloc(GenomeSize);
00166     for (int j = 0; j < GenomeSize; j++) {
00167         // calculating growth rate for a given gene for a current Environment value
00168         gsl_vector_set(TempVector, j, CellsGenotype.calculateGrowthFromOneGene(j, Env));
00169     }
00170     // picking the maximal value for given Env conditions
00171     CellsGrossGrowthRateAtTheMoment = gsl_vector_max(TempVector);
00172     gsl_vector_free(TempVector);

00224 void Cell::evaluateInternalResourcesCost(double gene_replication_cost,
00225         int core_genes, double metabolic_costs) {
00226     double GenomeSize = (double) CellsGenotype.GenotypeValues.size();
00227     double recentCostTemp;
00228     // sum of cost of living in one iteration
00229     recentCostTemp = -gene_replication_cost * ((double) core_genes \
00230         + GenomeSize + GenomeSize * GenomeSize) - metabolic_costs;
00231     if (recentCostTemp < -CellsResource) {
00232         recentLoss = -CellsResource;
00233         CellsResource = 0.0;
00234     }
00235     if (recentCostTemp >= -CellsResource) {
00236         recentLoss = recentCostTemp;
00237         CellsResource = CellsResource + recentCostTemp;
