00105         double gene_replication_cost, int core_genes, double metabolic_costs,
00106         double gain_from_the_genotype_ratio) {
00107     // number of cells in population
00108     int PopSize = (int) PopulationOfCells.size();
00109     rngEngine* pRandomNumberGenerator = rngEngine::getInstance();
00110     double SharedBetweenFactor = 0.0;
00111 
00112     if (mode_of_feeding.compare("p") == 0) {
00113         // array of the demands of all the cells in this round
00114         double CellsDemandsInThisRound[PopSize];
00115         // costs of living - cells lose resource and free resource increase
00116         for (int k = 0; k < PopSize; k++) {
00117             // evaluation of internal resources of a cell (just costs)
00118             PopulationOfCells[k].evaluateInternalResourcesCost(gene_replication_cost,
00119                     core_genes, metabolic_costs);
00120             // ageing cells
00121             PopulationOfCells[k].ageingACell();
00122             // evaluation of free resources
00123             EnvFreeResources -= PopulationOfCells[k].recentLoss;
00124             // zeroing cell's recent gain of resources
00125             PopulationOfCells[k].zeroRecentGainOfCell();
00126             // evaluation how much a cell can gain in this round of feeding
00127             PopulationOfCells[k].evaluateGrowthRateFromGenotypeOnePointEnv(Env);
00128             // Demand of a cell for resources
00129             CellsDemandsInThisRound[k] = gain_from_the_genotype_ratio \
00130                 * PopulationOfCells[k].CellsGrossGrowthRateAtTheMoment;
00131             SharedBetweenFactor += CellsDemandsInThisRound[k];
00132         }
00133         SharedBetweenFactor = EnvFreeResources / SharedBetweenFactor;
00134 
00135         // the feeding itself:
00136         // cells are being fed and Free Resources are being depleted
00137         for (int j = 0; j < PopSize; j++) {
00138             //evaluation of the gain of resources of a cell
00139             PopulationOfCells[j].evaluateInternalResourcesGainProportional
00140                     (gain_from_the_genotype_ratio, SharedBetweenFactor);
00141             // evaluation of free resources
00142             EnvFreeResources -= PopulationOfCells[j].recentGain;
00143         }
00144     }
00145     if (mode_of_feeding.compare("s") == 0) {
00146         // initialising a queuing array which will decide who will be fed first
00147         int ShuffleArr[PopSize];
00148         // filling a queuing array which will decide who will be fed first
00149         for (int i = 0; i < PopSize; i++) {
00150             ShuffleArr[i] = i;
00151         }
00152         // Shuffling elements of queuing array by randomly exchanging each with one other
00153         for (int i = 0; i < (PopSize - 1); i++) {
00154             // Random remaining position
00155             int r = i + (((int) \
00156                 (10000000.0 * pRandomNumberGenerator->rngMTgetDauble())) % (PopSize - i));
00157             int temp = ShuffleArr[i];
00158             ShuffleArr[i] = ShuffleArr[r];
00159             ShuffleArr[r] = temp;
00160         }
00161         // costs of living - cells lose resource and free resource increase
00162         for (int k = 0; k < PopSize; k++) {
00163             // evaluation of internal resources of a cell (just costs)
00164             PopulationOfCells[k].evaluateInternalResourcesCost(gene_replication_cost,
00165                     core_genes, metabolic_costs);
00166             // evaluation of free resources
00167             EnvFreeResources -= PopulationOfCells[k].recentLoss;
00168             // ageing cells
00169             PopulationOfCells[k].ageingACell();
00170             // zeroing cell's recent gain of resources
00171             PopulationOfCells[k].zeroRecentGainOfCell();
00172         }
00173         // the feeding itself:
00174         // cells are being fed and Free Resources are being depleted
00175         for (int j = 0; j < PopSize; j++) {
00176             if (EnvFreeResources >= PopulationOfCells[ShuffleArr[j]].CellsResource) {
00177                 // evaluation how much a cell can gain in this round of feeding
00178                 PopulationOfCells[ShuffleArr[j]].evaluateGrowthRateFromGenotypeOnePointEnv(Env);
00179                 //evaluation of the gain of resources of a cell
00180                 PopulationOfCells[ShuffleArr[j]].evaluateInternalResourcesGainSequential
00181                         (gain_from_the_genotype_ratio);
00182                 // evaluation of free resources
00183                 EnvFreeResources -= PopulationOfCells[ShuffleArr[j]].recentGain;
00184             } else {
00185                 break;
00186             }
00187         }
00188     }
00189     if ((mode_of_feeding.compare("p") != 0) && (mode_of_feeding.compare("s") != 0)) {

00191             << std::endl;

00247             PopulationOfCells.erase(PopulationOfCells.begin() + i);
00248             // count number of killed
00249             numberOfDeaths += 1;
00250         }
00251     }
00252     // let the fittest win! (reproduction and mutation)
00253     PopSize = (int) PopulationOfCells.size();
00254     // sequential feeding:
00255     if (mode_of_feeding.compare("s") == 0) {
00256         for (int j = 0; j < PopSize; j++) {
00257             PopulationOfCells[j].setCellReprodFlagFalse();

00276                 // mutate (if necessary) both cells
00277                 PopulationOfCells[j].mutateCell(Env, duplication_probability,
00278                         deletion_probability, mutate_threshold, minimum_genotype_size,
00279                         not_super_gene_param, gene_space_width, environment_resolution,
00280                         one_over_total_env_space_surf);
00281                 PopulationOfCells.back().mutateCell(Env, duplication_probability,
00282                         deletion_probability, mutate_threshold, minimum_genotype_size,
00283                         not_super_gene_param, gene_space_width, environment_resolution,

00361 void Ecosystem::horizontalGeneTransfer(double hgt_donor_prob,
00362     double hgt_gene_trans) {
00363     int LystSize = PopulationOfCells.size();
00364     int k;
00365     bool did_hgt_happened;
00366     // calling and initialising random number generator
00367     rngEngine* pRandomNumberGenerator = rngEngine::getInstance();
00368     Tagging_system* pTagging_system = Tagging_system::getInstance();
00369     for (int i = 0; i < LystSize; i++) {
00370         // tossing if a cell will be a donor
00371         if (pRandomNumberGenerator->rngMTgetDauble() < hgt_donor_prob){
00372             // picking an acceptor cell
00373             do {
00374                 k = (int) floor(pRandomNumberGenerator->rngMTgetDauble() \
00375                     * LystSize);
00376             } while ( k == i);
00377             int GenoypeSize = PopulationOfCells[i].CellsGenotype.GenotypeValues.size();
00378             did_hgt_happened = false;
00379             for (int j = 0; j < GenoypeSize; j++){
00380                 // tossing if a gene gets transfered 
00381                 if (pRandomNumberGenerator->rngMTgetDauble() < hgt_gene_trans){
00382                     PopulationOfCells[k].CellsGenotype.GenotypeValues.push_back
00383                     (PopulationOfCells[i].CellsGenotype.GenotypeValues[j]);
00384                     // incrementing the counter of accepted genes
00385                     PopulationOfCells[k].CellsGenotype.TotalHGT_number += 1;
00386                     did_hgt_happened = true;
00387                 }
00388                 if (did_hgt_happened == true){
00389                     PopulationOfCells[k].OwnTag = pTagging_system->getTag();
00390                 }
00391             }
00392         }
