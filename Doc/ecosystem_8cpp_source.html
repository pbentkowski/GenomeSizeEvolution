105 void Ecosystem::feedCells(std::string mode_of_feeding, double Env,
106  double gene_replication_cost, int core_genes, double metabolic_costs,
107  double gain_from_the_genotype_ratio) {
108  // number of cells in population
109  int PopSize = (int) PopulationOfCells.size();
110  rngEngine* pRandomNumberGenerator = rngEngine::getInstance();
111  double SharedBetweenFactor = 0.0;
112 
113  if (mode_of_feeding.compare("p") == 0) {
114  // array of the demands of all the cells in this round
115  double CellsDemandsInThisRound[PopSize];
116  // costs of living - cells lose resource and free resource increase
117  for (int k = 0; k < PopSize; k++) {
118  // evaluation of internal resources of a cell (just costs)
119  PopulationOfCells[k].evaluateInternalResourcesCost(gene_replication_cost,
120  core_genes, metabolic_costs);
121  // ageing cells
122  PopulationOfCells[k].ageingACell();
123  // evaluation of free resources
124  EnvFreeResources -= PopulationOfCells[k].recentLoss;
125  // zeroing cell's recent gain of resources
126  PopulationOfCells[k].zeroRecentGainOfCell();
127  // evaluation how much a cell can gain in this round of feeding
128  PopulationOfCells[k].evaluateGrowthRateFromGenotypeOnePointEnv(Env);
129  // Demand of a cell for resources
130  CellsDemandsInThisRound[k] = gain_from_the_genotype_ratio \
131  * PopulationOfCells[k].CellsGrossGrowthRateAtTheMoment;
132  SharedBetweenFactor += CellsDemandsInThisRound[k];
133  }
134  SharedBetweenFactor = EnvFreeResources / SharedBetweenFactor;
135 
136  // the feeding itself:
137  // cells are being fed and Free Resources are being depleted
138  for (int j = 0; j < PopSize; j++) {
139  //evaluation of the gain of resources of a cell
140  PopulationOfCells[j].evaluateInternalResourcesGainProportional
141  (gain_from_the_genotype_ratio, SharedBetweenFactor);
142  // evaluation of free resources
143  EnvFreeResources -= PopulationOfCells[j].recentGain;
144  }
145  }
146  if (mode_of_feeding.compare("s") == 0) {
147  // initialising a queuing array which will decide who will be fed first
148  int ShuffleArr[PopSize];
149  // filling a queuing array which will decide who will be fed first
150  for (int i = 0; i < PopSize; i++) {
151  ShuffleArr[i] = i;
152  }
153  // Shuffling elements of queuing array by randomly exchanging each with one other
154  for (int i = 0; i < (PopSize - 1); i++) {
155  // Random remaining position
156  int r = i + (((int) \
157  (10000000.0 * pRandomNumberGenerator->rngMTgetDauble())) % (PopSize - i));
158  int temp = ShuffleArr[i];
159  ShuffleArr[i] = ShuffleArr[r];
160  ShuffleArr[r] = temp;
161  }
162  // costs of living - cells lose resource and free resource increase
163  for (int k = 0; k < PopSize; k++) {
164  // evaluation of internal resources of a cell (just costs)
165  PopulationOfCells[k].evaluateInternalResourcesCost(gene_replication_cost,
166  core_genes, metabolic_costs);
167  // evaluation of free resources
168  EnvFreeResources -= PopulationOfCells[k].recentLoss;
169  // ageing cells
170  PopulationOfCells[k].ageingACell();
171  // zeroing cell's recent gain of resources
172  PopulationOfCells[k].zeroRecentGainOfCell();
173  }
174  // the feeding itself:
175  // cells are being fed and Free Resources are being depleted
176  for (int j = 0; j < PopSize; j++) {
177  if (EnvFreeResources >= PopulationOfCells[ShuffleArr[j]].CellsResource) {
178  // evaluation how much a cell can gain in this round of feeding
179  PopulationOfCells[ShuffleArr[j]].evaluateGrowthRateFromGenotypeOnePointEnv(Env);
180  //evaluation of the gain of resources of a cell
181  PopulationOfCells[ShuffleArr[j]].evaluateInternalResourcesGainSequential
182  (gain_from_the_genotype_ratio);
183  // evaluation of free resources
184  EnvFreeResources -= PopulationOfCells[ShuffleArr[j]].recentGain;
185  } else {
186  break;
187  }
188  }
189  }
190  if ((mode_of_feeding.compare("p") != 0) && (mode_of_feeding.compare("s") != 0)) {
191  std::cout << "Mode can be only p (proportional) or s (sequential)"\
192  << std::endl;
193  }
194 }

221 void Ecosystem::reproduceOrKillCell(std::string mode_of_feeding, double Env,
222  double duplication_probability, double deletion_probability,
223  double mutate_threshold, int minimum_genotype_size, double not_super_gene_param,
224  double gene_space_width, double environment_resolution,
225  double one_over_total_env_space_surf, double random_death_rate,
226  double minimal_cells_resource, double reproduction_resourse_size,
227  double disparity_at_reproduction) {
228  // calling and initialising random number generator
229  rngEngine* pRandomNumberGenerator = rngEngine::getInstance();
230  numberOfDeaths = 0;
231  numberOfBirths = 0;
232  int PopSize;
233  // kill cells 'by accident' (random death)
234  PopSize = (int) PopulationOfCells.size();
235  for (int i = PopSize - 1; i >= 0; i--) {
236  if ((pRandomNumberGenerator->rngMTgetDauble()) < random_death_rate) {
237  EnvFreeResources = EnvFreeResources + PopulationOfCells[i].CellsResource;
238  PopulationOfCells.erase(PopulationOfCells.begin() + i);
239  // count number of killed
240  numberOfDeaths += 1;
241  }
242  }
243  // kill the weak ones
244  PopSize = (int) PopulationOfCells.size();
245  for (int i = PopSize - 1; i >= 0; i--) {
246  if ((PopulationOfCells[i].CellsResource) < minimal_cells_resource) {
247  EnvFreeResources = EnvFreeResources + PopulationOfCells[i].CellsResource;
248  PopulationOfCells.erase(PopulationOfCells.begin() + i);
249  // count number of killed
250  numberOfDeaths += 1;
251  }
252  }
253  // let the fittest win! (reproduction and mutation)
254  PopSize = (int) PopulationOfCells.size();
255  // sequential feeding:
256  if (mode_of_feeding.compare("s") == 0) {
257  for (int j = 0; j < PopSize; j++) {
258  PopulationOfCells[j].setCellReprodFlagFalse();
259  if (PopulationOfCells[j].CellsResource > reproduction_resourse_size) {
260  // calculating the intake ratio for the reproducing cell
261  PopulationOfCells[j].setAgeSinceLastReprod();
262  PopulationOfCells[j].setAvarageIntakeTillRepr();
263  PopulationOfCells[j].setAvarageTotalGainTillRepr();
264  // Dividing the resource between the mother and the offspring
265  PopulationOfCells[j].CellsResource = \
266  (PopulationOfCells[j].CellsResource) * 0.5;
267  // note the fact this cell has reproduced
268  PopulationOfCells[j].setCellReprodFlagTrue();
269  PopulationOfCells[j].incrementNumberOfReprod();
270  PopulationOfCells[j].setCellsResAfterReprod();
271  // create a 'newborn' cell
272  PopulationOfCells.push_back(PopulationOfCells[j]);
273  // set new cell's params to zero (ones which need to be zeroed)
274  PopulationOfCells.back().resetParamsOfNewCell();
275  // inheriting mothers tag
276  PopulationOfCells.back().MothersTag = PopulationOfCells[j].OwnTag;
277  // mutate (if necessary) both cells
278  PopulationOfCells[j].mutateCell(Env, duplication_probability,
279  deletion_probability, mutate_threshold, minimum_genotype_size,
280  not_super_gene_param, gene_space_width, environment_resolution,
281  one_over_total_env_space_surf);
282  PopulationOfCells.back().mutateCell(Env, duplication_probability,
283  deletion_probability, mutate_threshold, minimum_genotype_size,
284  not_super_gene_param, gene_space_width, environment_resolution,
285  one_over_total_env_space_surf);
286  // count number of born to record cell's age at first reproduction
287  numberOfBirths += 1;
288  }
289  }
290  }
291  // proportional feeding:
292  if (mode_of_feeding.compare("p") == 0) {
293  double Division_disparity;
294  for (int j = 0; j < PopSize; j++) {
295  PopulationOfCells[j].setCellReprodFlagFalse();
296  if (PopulationOfCells[j].CellsResource > reproduction_resourse_size) {
297  PopulationOfCells[j].setAgeSinceLastReprod();
298  PopulationOfCells[j].setAvarageIntakeTillRepr();
299  PopulationOfCells[j].setAvarageTotalGainTillRepr();
300  Division_disparity = 0.5 + disparity_at_reproduction \
301  * pRandomNumberGenerator->rngMTgetDauble();
302  // note the fact this cell has reproduced
303  PopulationOfCells[j].setCellReprodFlagTrue();
304  PopulationOfCells[j].incrementNumberOfReprod();
305  // create a 'newborn' cell
306  PopulationOfCells.push_back(PopulationOfCells[j]);
307  PopulationOfCells.back().CellsResource = (1.0 - Division_disparity) \
308  * (PopulationOfCells[j].CellsResource);
309  PopulationOfCells[j].CellsResource = Division_disparity \
310  * (PopulationOfCells[j].CellsResource);
311  PopulationOfCells[j].setCellsResAfterReprod();
312  // set new cell's params to zero
313  PopulationOfCells.back().resetParamsOfNewCell();
314  // inheriting mothers tag
315  PopulationOfCells.back().MothersTag = PopulationOfCells[j].OwnTag;
316  // mutate (if necessary) both cells
317  PopulationOfCells[j].mutateCell(Env, duplication_probability,
318  deletion_probability, mutate_threshold, minimum_genotype_size,
319  not_super_gene_param, gene_space_width, environment_resolution,
320  one_over_total_env_space_surf);
321  PopulationOfCells.back().mutateCell(Env, duplication_probability,
322  deletion_probability, mutate_threshold, minimum_genotype_size,
323  not_super_gene_param, gene_space_width, environment_resolution,
324  one_over_total_env_space_surf);
325  // count number of born
326  numberOfBirths += 1;
327  }
328  }
329  }

362 void Ecosystem::horizontalGeneTransfer(double hgt_donor_prob,
363  double hgt_gene_trans) {
364  int LystSize = PopulationOfCells.size();
365  int k;
366  bool did_hgt_happened;
367  // calling and initialising random number generator
368  rngEngine* pRandomNumberGenerator = rngEngine::getInstance();
369  Tagging_system* pTagging_system = Tagging_system::getInstance();
370  for (int i = 0; i < LystSize; i++) {
371  // tossing if a cell will be a donor
372  if (pRandomNumberGenerator->rngMTgetDauble() < hgt_donor_prob){
373  // picking an acceptor cell
374  do {
375  k = (int) floor(pRandomNumberGenerator->rngMTgetDauble() \
376  * LystSize);
377  } while ( k == i);
378  int GenoypeSize = PopulationOfCells[i].CellsGenotype.GenotypeValues.size();
379  did_hgt_happened = false;
380  for (int j = 0; j < GenoypeSize; j++){
381  // tossing if a gene gets transfered
382  if (pRandomNumberGenerator->rngMTgetDauble() < hgt_gene_trans){
383  PopulationOfCells[k].CellsGenotype.GenotypeValues.push_back
384  (PopulationOfCells[i].CellsGenotype.GenotypeValues[j]);
385  // incrementing the counter of accepted genes
386  PopulationOfCells[k].CellsGenotype.TotalHGT_number += 1;
387  did_hgt_happened = true;
388  }
389  if (did_hgt_happened == true){
390  PopulationOfCells[k].OwnTag = pTagging_system->getTag();
391  }
392  }
393  }
