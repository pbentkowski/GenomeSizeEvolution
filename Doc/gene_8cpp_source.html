00043 void Gene::setGeneValues(double not_super_gene_param, double gene_space_width) {
00044     // GeneValue[0] - hight ; GeneValue[1] - variance ; GeneValue[2] - mean
00045     // of the Gaussian function

00047     // hight [0,1)

00049     // variance; it scales to hight to lover the probability of emergence
00050     // of a 'supergene'

00053     GeneValue[2] = 2.0 * gene_space_width * p_rngEngine->rngMTgetDauble() \
00054         - gene_space_width;
