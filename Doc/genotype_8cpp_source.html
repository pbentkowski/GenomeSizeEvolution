86 void Genotype::deleteGenes(double deletion_probability, int minimum_genotype_size) {
87  // calling and initialising random number generator
88  rngEngine* p_rngEngine = rngEngine::getInstance();
89  //measuring the size of genotype
90  int Size = (int) GenotypeValues.size();
91  // marker of gene deletion
92  bool TEST_IF_DELETED = false;
93  // counter prevents loosing all the genes (and segmentation faults as a result)
94  int counter = Size - minimum_genotype_size - 1;
95  // calculating what's the probability of deletion of particular gene
96  for (int i = Size - 1; i >= 0; i--) {
97  if ((p_rngEngine->rngMTgetDauble() < deletion_probability) && (counter >= 0)) {
98  GenotypeValues.erase(GenotypeValues.begin() + i);
99  TEST_IF_DELETED = true;
100  counter -= 1;
101  TotalDeletionNumber += 1;
102  }
103  }
104  // if TEST_IF_DELETED is false it means that there were no deletions,
105  // so the first gene is being deleted by force
106  if ((TEST_IF_DELETED == false) && (counter >= 0)) {
107  GenotypeValues.erase(GenotypeValues.begin());
108  TotalDeletionNumber += 1;
109  }
110  // if TEST_IF_DELETED is true it means that there were deletions already,
111  // so a dice can be thrown to decide if the first gene will be deleted
112  if ((TEST_IF_DELETED == true) && (p_rngEngine->rngMTgetDauble() \
113  < deletion_probability) && (counter >= 0)) {
114  GenotypeValues.erase(GenotypeValues.begin());
115  TotalDeletionNumber += 1;
116  }
117  TEST_IF_DELETED = false;
118 }

287 void Genotype::evaluateAdaptiveLandcape(double environment_resolution,
288  double one_over_total_env_space_surf) {
289  // measuring genome size and env resolution
290  int GenomeSize = (int) GenotypeValues.size();
291  long int GLsize = (long int) (2.0 / environment_resolution);
292  // env value
293  double env = -1.0;
294  // surface
295  double s = 0.0;
296  MaxGrowthRate = 0.0;
297  double maxGeneValue;
298  // Temporary vector collecting Gross Growth Rates provided by all the
299  // genes for a given Environment conditions
300  gsl_vector * TempVector = gsl_vector_alloc(GenomeSize);
301  for (int e = 0; e < GLsize; e++) { // while (env <= 1.0)
302  for (int j = 0; j < GenomeSize; j++) {
303  // calculating growth rate for a given gene in a given Environment value
304  gsl_vector_set(TempVector, j, calculateGrowthFromOneGene(j, env));
305  }
306  maxGeneValue = gsl_vector_max(TempVector);
307  GenomeLandscape[e] = maxGeneValue;
308  if (maxGeneValue > MaxGrowthRate) {
309  // finding the global maximum of the genotype's envelope
310  MaxGrowthRate = maxGeneValue;
311  }
312  // calculating the surface of one bin using highest gene value and
313  // incrementing total surface
314  s = s + maxGeneValue * environment_resolution;
315  env = env + environment_resolution;
316  }
317  EnvSpaceFraction = s * one_over_total_env_space_surf;
318  gsl_vector_free(TempVector);
319 }
