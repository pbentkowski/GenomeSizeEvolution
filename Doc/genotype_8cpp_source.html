00082 void Genotype::deleteGenes(double deletion_probability, int minimum_genotype_size) {
00083     // calling and initialising random number generator
00084     rngEngine* p_rngEngine = rngEngine::getInstance();
00085     //measuring the size of genotype
00086     int Size = (int) GenotypeValues.size();
00087     // marker of gene deletion
00088     bool TEST_IF_DELETED = false;
00089     // counter prevents loosing all the genes (and segmentation faults as a result)
00090     int counter = Size - minimum_genotype_size - 1;
00091     // calculating what's the probability of deletion of particular gene
00092     for (int i = Size - 1; i >= 0; i--) {
00093         if ((p_rngEngine->rngMTgetDauble() < deletion_probability) && (counter >= 0)) {
00094             GenotypeValues.erase(GenotypeValues.begin() + i);
00095             TEST_IF_DELETED = true;
00096             counter -= 1;
00097             TotalDeletionNumber += 1;
00098         }
00099     }
00100     // if TEST_IF_DELETED is false it means that there were no deletions,
00101     // so the first gene is being deleted by force
00102     if ((TEST_IF_DELETED == false) && (counter >= 0)) {
00103         GenotypeValues.erase(GenotypeValues.begin());
00104         TotalDeletionNumber += 1;
00105     }
00106     // if TEST_IF_DELETED is true it means that there were deletions already,
00107     // so a dice can be thrown to decide if the first gene will be deleted
00108     if ((TEST_IF_DELETED == true) && (p_rngEngine->rngMTgetDauble() \
00109             < deletion_probability) && (counter >= 0)) {
00110         GenotypeValues.erase(GenotypeValues.begin());
00111         TotalDeletionNumber += 1;
00112     }
00113     TEST_IF_DELETED = false;
00114 }

00282 void Genotype::evaluateAdaptiveLandcape(double environment_resolution,
00283         double one_over_total_env_space_surf) {
00284     // measuring genome size
00285     int GenomeSize = (int) GenotypeValues.size();
00286     // env value
00287     double env = 0.005;
00288     // surfase
00289     double s = 0.0;
00290     MaxGrowthRate = 0.0;
00291     double maxGeneValue;
00292     // Temporary vector collecting Gross Growth Rates provided by all the
00293     // genes for a given Environment conditions
00294     gsl_vector * TempVector = gsl_vector_alloc(GenomeSize);
00295     while (env < 1.0) {
00296         for (int j = 0; j < GenomeSize; j++) {
00297             // calculating growth rate for a given gene in a given Environment value
00298             gsl_vector_set(TempVector, j, calculateGrowthFromOneGene(j, env));
00299         }
00300         maxGeneValue = gsl_vector_max(TempVector);
00301         if (maxGeneValue > MaxGrowthRate) {
00302             // finding the global maximum of the genotype's envelope
00303             MaxGrowthRate = maxGeneValue;
00304         }
00305         // calculating the surface of one bin using highest gene value and
00306         // incrementing total surface
00307         s = s + maxGeneValue * environment_resolution;
00308         env = env + environment_resolution;
00309     }
00310     EnvSpaceFraction = s * one_over_total_env_space_surf;
00311     gsl_vector_free(TempVector);
00312 }
