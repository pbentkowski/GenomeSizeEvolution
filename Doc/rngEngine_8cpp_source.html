92 void rngEngine::rngMTset(unsigned long int s) {
93  if (s == 0) {
94  // the default seed is 4357
95  s = 4357;
96  }
97  mt[0] = s & 0xffffffffUL;
98  for (mti = 1; mti < N_rng; mti++) {
99  // See Knuth's "Art of Computer Programming" Vol. 2, 3rd Ed. p.106
100  // for multiplier.
101  mt[mti] = (1812433253UL * (mt[mti - 1] ^ (mt[mti - 1] >> 30)) + mti);
102  mt[mti] &= 0xffffffffUL;
103  }
104 }

111 unsigned long rngEngine::rngMTget() {
112  unsigned long y;
113 #define MAGIC(y) (((y)&0x1) ? 0x9908b0dfUL : 0)
114  if (mti >= N_rng) {
115  // generate N words at one time
116  int kk;
117  // if rngMTset() has not been called, a default initial seed is used
118  if (mti == (N_rng + 1)) {
119  rngMTset(5489UL);
120  }
121  for (kk = 0; kk < (N_rng - M_rng); kk++) {
122  y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
123  mt[kk] = mt[kk + M_rng ] ^ (y >> 1) ^ MAGIC(y);
124  }
125  for (; kk < N_rng - 1; kk++) {
126  y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
127  mt[kk] = mt[kk + (M_rng - N_rng)] ^ (y >> 1) ^ MAGIC(y);
128  }
129  y = (mt[N_rng - 1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
130  mt[N_rng - 1] = mt[M_rng - 1] ^ (y >> 1) ^ MAGIC(y);
131 
132  mti = 0;
133  }
134  // Tempering
135  y = mt[mti++];
136  y ^= (y >> 11);
137  y ^= (y << 7) & 0x9d2c5680UL;
138  y ^= (y << 15) & 0xefc60000UL;
139  y ^= (y >> 18);
140 
141  return y;
142 }
