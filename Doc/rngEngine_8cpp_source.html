00092 void rngEngine::rngMTset(unsigned long int s) {
00093     if (s == 0) {
00094         // the default seed is 4357
00095         s = 4357;
00096     }
00097     mt[0] = s & 0xffffffffUL;
00098     for (mti = 1; mti < N_rng; mti++) {
00099         // See Knuth's "Art of Computer Programming" Vol. 2, 3rd Ed. p.106
00100         // for multiplier.
00101         mt[mti] = (1812433253UL * (mt[mti - 1] ^ (mt[mti - 1] >> 30)) + mti);
00102         mt[mti] &= 0xffffffffUL;
00103     }
00104 }

00111 unsigned long rngEngine::rngMTget() {
00112     unsigned long y;
00113 #define MAGIC(y) (((y)&0x1) ? 0x9908b0dfUL : 0)
00114     if (mti >= N_rng) {
00115         // generate N words at one time
00116         int kk;
00117         // if rngMTset() has not been called, a default initial seed is used
00118         if (mti == (N_rng + 1)) {
00119             rngMTset(5489UL);
00120         }
00121         for (kk = 0; kk < (N_rng - M_rng); kk++) {
00122             y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
00123             mt[kk] = mt[kk + M_rng ] ^ (y >> 1) ^ MAGIC(y);
00124         }
00125         for (; kk < N_rng - 1; kk++) {
00126             y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
00127             mt[kk] = mt[kk + (M_rng - N_rng)] ^ (y >> 1) ^ MAGIC(y);
00128         }
00129         y = (mt[N_rng - 1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
00130         mt[N_rng - 1] = mt[M_rng - 1] ^ (y >> 1) ^ MAGIC(y);
00131 
00132         mti = 0;
00133     }
00134     // Tempering 
00135     y = mt[mti++];
00136     y ^= (y >> 11);
00137     y ^= (y << 7) & 0x9d2c5680UL;
00138     y ^= (y << 15) & 0xefc60000UL;
00139     y ^= (y >> 18);
00140 
00141     return y;
00142 }
